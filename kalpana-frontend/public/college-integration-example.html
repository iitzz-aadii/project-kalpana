üíª Integration Code

Here's how this integration was implemented on our website:

1. Include the SDK

<script src="https://app.kalpana-timetable.com/kalpana-sdk.js"></script>

2. Initialize the SDK

<script> // Initialize Kalpana SDK const kalpana = new KalpanaSDK({ apiBaseUrl: 'https://api.kalpana-timetable.com', frontendUrl: 'https://app.kalpana-timetable.com', theme: 'light', primaryColor: '#667eea', secondaryColor: '#764ba2', language: 'en', debug: true }); // Initialize with college ID kalpana.init('nit-srinagar-001').then(result => { if (result.success) { console.log('Kalpana SDK initialized successfully'); // Create timetable widget kalpana.createTimetableWidget('timetable-container', { height: '600px', showControls: true, allowFullscreen: true }); // Create faculty widget kalpana.createFacultyWidget('faculty-container', 'faculty-123', { height: '400px', showPersonalInfo: true }); // Create student login widget kalpana.createLoginWidget('student-container', { height: '500px', showRegistration: true, allowGoogleLogin: true }); } else { console.error('Failed to initialize SDK:', result.error); } }); </script>

3. Alternative: HTML Data Attributes

<!-- Timetable Widget --> <div id="timetable-widget" data-kalpana-widget="timetable" data-college-id="nit-srinagar-001" data-options='{"height": "600px", "showControls": true}' ></div> <!-- Faculty Widget --> <div id="faculty-widget" data-kalpana-widget="faculty" data-college-id="nit-srinagar-001" data-faculty-id="faculty-123" data-options='{"height": "400px"}' ></div>
